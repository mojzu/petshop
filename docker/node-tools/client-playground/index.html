Client Playground

The variables httpClient and grpcClient are defined globally, use them in the web console

Or when using dev-client-playground changes to files in docker/node-tools/client-playground will be reloaded live

Example 1 - PetPut

httpClient

req = {
    id: 32,
    name: "Name1",
    category: {id: 23, name: "Cat1"},
    photoUrls: ["Photo1"],
    tags: [{id: 45, name: "Tag1"}], status: "PENDING"
};
await httpClient.petshopPetPut(req);

grpcClient

category = (new Category()).setId(23).setName("Category1");
tag = (new Tag()).setId(45).setName("Tag1");
req = (new Pet())
    .setId(32)
    .setName("Name1")
    .setStatus(Status.PENDING)
    .setCategory(category)
    .addPhotoUrls("PhotoUrl1")
    .setTagsList([tag]);
await grpcClient.petPut(req);

Example 2 - PetFindByStatus

httpClient

req = {status: "PENDING"};
await httpClient.petshopPetFindByStatus(req);

grpcClient

req = (new FindByStatus()).addStatus(Status.PENDING);
await grpcClient.petFindByStatus(req);

Example 3 - PetFindByTag

httpClient

req = {tags: ["Tag1", "Tag2"]};
await httpClient.petshopPetFindByTag(req);

grpcClient

req = (new FindByTag()).setTagsList(["Tag1", "Tag2"]);
await grpcClient.petFindByTag(req);

Example 4 - HttpBody

httpClient

This response will be converted to text/html 
await httpClient.petshopHttpBodyEx({});

grpcClient

This response will not be converted 
req = new Empty();
await grpcClient.httpBodyEx(req);

Example 5 - Json

httpClient

req = { arbitraryJson: true, nestedObject: { sub: "string" } };
await httpClient.petshopJsonEx(req);

grpcClient

req = Struct.fromJavaScript({ arbitraryJson: true, nestedObject: { sub: "string" } });
await grpcClient.jsonEx(req);

Example 6 - Validation

httpClient

req = { email: "notavalidemail", name: "validname" };
await httpClient.petshopValidationEx(req);
req = { email: "validemail@example.com", name: "validname" };
await httpClient.petshopValidationEx(req);

grpcClient

req = (new User()).setEmail("notavalidemail").setName("validname");
await grpcClient.validationEx(req);
req = (new User()).setEmail("validemail@example.com").setName("validname");
await grpcClient.validationEx(req);

Example 7 - Streaming

httpClient

await httpClient.petshopStreamingEx({message:"message1"});

grpcClient

req = (new Echo()).setMessage("message1");
stream = grpcClient.streamingEx(req);
stream.on('data', (res) => console.log("stream message", res.getMessage()));
stream.on('status', (status) => console.log("stream status", status));
stream.on('end', (end) => console.log("stream end"));

Example 8 - Authentication

This only works with the authz example

httpClient

await httpClient.petshopAuthenticationRequiredEx({});

grpcClient

req = new Empty();
await grpcClient.authenticationRequiredEx(req);
